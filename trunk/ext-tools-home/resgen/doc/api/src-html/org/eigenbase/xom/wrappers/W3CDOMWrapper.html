026    package org.eigenbase.xom.wrappers;
027    
028    import org.eigenbase.xom.DOMWrapper;
029    import org.eigenbase.xom.XOMUtil;
030    import org.w3c.dom.CharacterData;
031    import org.w3c.dom.*;
032    
033    /**
034     * This implementation of DOMWrapper wraps any w3c DOM-compliant java
035     * XML Parser.
036     */
037    public class W3CDOMWrapper implements DOMWrapper {
038    
039        private Node node;
040    
041        /**
042         * W3CDOMWrapper parses XML based on a Node.  The Node may be either an
043         * Element or some form of text node.
044         */
045        public W3CDOMWrapper(Node node)
046        {
047            this.node = node;
048        }
049    
050        /**
051         * Map the Node's type to DOMWrapper's simplified concept of type.
052         */
053        public int getType()
054        {
055            int nodeType = node.getNodeType();
056            switch (nodeType) {
057            case Node.ELEMENT_NODE:
058                return ELEMENT;
059            case Node.COMMENT_NODE:
060                return COMMENT;
061            case Node.CDATA_SECTION_NODE:
062                return CDATA;
063            case Node.TEXT_NODE:
064                return FREETEXT;
065            default:
066                return UNKNOWN;
067            }
068        }
069    
070        /**
071         * Retrieve the tag name directly.  Return null immediately if not an
072         * element.
073         **/
074        public String getTagName()
075        {
076            if(getType() != ELEMENT)
077                return null;
078            return ((Element)node).getTagName();
079        }
080    
081        /**
082         * Return the attribute.  Return null if the attribute isn't defined,
083         * or if not an element.  This behavior differs from the underlying DOM,
084         * which returns an empty string for undefined attributes.
085         */
086        public String getAttribute(String attrName)
087        {
088            if(getType() != ELEMENT)
089                return null;
090            String attrVal = ((Element)node).getAttribute(attrName);
091            if(attrVal == null || attrVal.length() == 0)
092                return null;
093            else
094                return attrVal;
095        }
096    
097        // implement DOMWrapper
098        public String[] getAttributeNames()
099        {
100            NamedNodeMap map = node.getAttributes();
101            int count = map.getLength();
102            String[] attributeNames = new String[count];
103            for (int i = 0; i < count; i++) {
104                attributeNames[i] = map.item(i).getLocalName();
105            }
106            return attributeNames;
107        }
108    
109        /**
110         * Recursively unwrap and create the contained text.  If the node is a
111         * comment, return the comment text; but ignore comments inside elements.
112         **/
113        public String getText()
114        {
115            if (node instanceof Comment) {
116                return ((Comment)node).getData();
117            } else {
118                StringBuffer sbuf = new StringBuffer();
119                appendNodeText(node, sbuf);
120                return sbuf.toString();
121            }
122        }
123    
124        // implement DOMWrapper
125        public String toXML()
126        {
127            boolean onlyElements = false;
128            return XOMUtil.wrapperToXml(this, onlyElements);
129        }
130    
131        /**
132         * Helper to collect all Text nodes into a buffer.
133         */
134        private static void appendNodeText(Node node, StringBuffer sbuf)
135        {
136            if (node instanceof Comment) {
137                // ignore it
138            }
139            else if (node instanceof CharacterData) {
140                // Text
141                sbuf.append(((CharacterData)node).getData());
142            } else if (node instanceof Element) {
143                NodeList nodeList = node.getChildNodes();
144                for(int i=0; i<nodeList.getLength(); i++)
145                    appendNodeText(nodeList.item(i), sbuf);
146            }
147        }
148    
149        /**
150         * Retrieve all children, and build an array of W3CDOMWrappers around
151         * each child that is of TEXT or ELEMENT type to return.
152         */
153        public DOMWrapper[] getChildren()
154        {
155            if(getType() != ELEMENT)
156                return new DOMWrapper[0];
157    
158            NodeList nodeList = node.getChildNodes();
159    
160            // Count the elements that are TEXT or ELEMENTs.
161            int count = 0;
162            for(int i=0; i<nodeList.getLength(); i++) {
163                Node nextNode = nodeList.item(i);
164                if(nextNode instanceof Element || nextNode instanceof Text)
165                    count++;
166            }
167    
168            // Create and populate the array
169            DOMWrapper[] ret = new DOMWrapper[count];
170            count = 0;
171            for(int i=0; i<nodeList.getLength(); i++) {
172                Node nextNode = nodeList.item(i);
173                if(nextNode instanceof Element || nextNode instanceof Text)
174                    ret[count++] = new W3CDOMWrapper(nextNode);
175            }
176    
177            // Done.
178            return ret;
179        }
180    
181        /**
182         * Retrieve all children, and build an array of W3CDOMWrappers around
183         * each ELEMENT child.
184         */
185        public DOMWrapper[] getElementChildren()
186        {
187            if(getType() != ELEMENT)
188                return new DOMWrapper[0];
189    
190            NodeList nodeList = node.getChildNodes();
191    
192            // Count the elements that are TEXT or ELEMENTs.
193            int count = 0;
194            for(int i=0; i<nodeList.getLength(); i++) {
195                Node nextNode = nodeList.item(i);
196                if(nextNode instanceof Element)
197                    count++;
198            }
199    
200            // Create and populate the array
201            DOMWrapper[] ret = new DOMWrapper[count];
202            count = 0;
203            for(int i=0; i<nodeList.getLength(); i++) {
204                Node nextNode = nodeList.item(i);
205                if(nextNode instanceof Element)
206                    ret[count++] = new W3CDOMWrapper(nextNode);
207            }
208    
209            // Done.
210            return ret;
211        }
212    
213    }
