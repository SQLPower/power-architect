032    /**
033     * Ant task which processes an XML file and generates a C++ or Java class from
034     * the resources in it.
035     *
036     * @author jhyde
037     * @since 19 September, 2005
038     * @version $Id: //open/util/resgen/src/org/eigenbase/resgen/XmlFileTask.java#10 $
039     */
040    class XmlFileTask extends FileTask
041    {
042        final String baseClassName;
043        final String cppBaseClassName;
044    
045        XmlFileTask(ResourceGenTask.Include include, String fileName,
046                  String className, String baseClassName, boolean outputJava,
047                  String cppClassName, String cppBaseClassName, boolean outputCpp)
048        {
049            this.include = include;
050            this.fileName = fileName;
051            this.outputJava = outputJava;
052            if (className == null) {
053                className = Util.fileNameToClassName(fileName, ".xml");
054            }
055            this.className = className;
056            if (baseClassName == null) {
057                baseClassName = "org.eigenbase.resgen.ShadowResourceBundle";
058            }
059            this.baseClassName = baseClassName;
060    
061            this.outputCpp = outputCpp;
062            if (cppClassName == null) {
063                cppClassName = Util.fileNameToCppClassName(fileName, ".xml");
064            }
065            this.cppClassName = cppClassName;
066            if (cppBaseClassName == null) {
067                cppBaseClassName = "ResourceBundle";
068            }
069            this.cppBaseClassName = cppBaseClassName;
070        }
071    
072        void process(ResourceGen generator) throws IOException {
073            URL url = Util.convertPathToURL(getFile());
074            ResourceDef.ResourceBundle resourceList = Util.load(url);
075            if (resourceList.locale == null) {
076                throw new BuildException(
077                        "Resource file " + url + " must have locale");
078            }
079    
080            ArrayList localeNames = new ArrayList();
081            if (include.root.locales == null) {
082                localeNames.add(resourceList.locale);
083            } else {
084                StringTokenizer tokenizer = new StringTokenizer(include.root.locales,",");
085                while (tokenizer.hasMoreTokens()) {
086                    String token = tokenizer.nextToken();
087                    localeNames.add(token);
088                }
089            }
090            
091            if (!localeNames.contains(resourceList.locale)) {
092                throw new BuildException(
093                        "Resource file " + url + " has locale '" +
094                        resourceList.locale +
095                        "' which is not in the 'locales' list");
096            }
097    
098            Locale[] locales = new Locale[localeNames.size()];
099            for (int i = 0; i < locales.length; i++) {
100                String localeName = (String) localeNames.get(i);
101                locales[i] = Util.parseLocale(localeName);
102                if (locales[i] == null) {
103                    throw new BuildException(
104                            "Invalid locale " + localeName);
105                }
106            }
107    
108    
109            if (outputJava) {
110                generateJava(generator, resourceList, null);
111            }
112    
113            generateProperties(generator, resourceList, null);
114    
115            for (int i = 0; i < locales.length; i++) {
116                Locale locale = locales[i];
117                if (outputJava) {
118                    generateJava(generator, resourceList, locale);
119                }
120                generateProperties(generator, resourceList, locale);
121            }
122    
123            if (outputCpp) {
124                generateCpp(generator, resourceList);
125            }
126        }
127    
128        private void generateProperties(
129                ResourceGen generator,
130                ResourceDef.ResourceBundle resourceList,
131                Locale locale) {
132            String fileName = Util.getClassNameSansPackage(className, locale) + ".properties";
133            File file = new File(getResourceDirectory(), fileName);
134            File srcFile = locale == null ?
135                getFile() :
136                new File(getSrcDirectory(), fileName);
137            if (file.exists()) {
138                if (locale != null) {
139                    if (file.equals(srcFile)) {
140                        // The locale.properties file already exists, and the
141                        // source and target locale.properties files are the
142                        // same. No need to create it, or even to issue a warning.
143                        // We were only going to create an empty file, anyway.
144                        return;
145                    }
146                }
147                if (file.lastModified() >= srcFile.lastModified()) {
148                    generator.comment(file + " is up to date");
149                    return;
150                }
151                if (!file.canWrite()) {
152                    generator.comment(file + " is read-only");
153                    return;
154                }
155            }
156            generator.comment("Generating " + file);
157            final FileOutputStream out;
158            try {
159                if (file.getParentFile() != null) {
160                    file.getParentFile().mkdirs();
161                }
162                out = new FileOutputStream(file);
163            } catch (FileNotFoundException e) {
164                throw new BuildException("Error while writing " + file, e);
165            }
166            PrintWriter pw = new PrintWriter(out);
167            try {
168                if (locale == null) {
169                    generateBaseProperties(resourceList, pw);
170                } else {
171                    generateProperties(pw, file, srcFile, locale);
172                }
173            } finally {
174                pw.close();
175            }
176        }

179        /**
180         * Generates a properties file containing a line for each resource.
181         */
182        private void generateBaseProperties(
183                ResourceDef.ResourceBundle resourceList, PrintWriter pw) {
184            String fullClassName = getClassName(null);
185            pw.println("# This file contains the resources for");
186            pw.println("# class '" + fullClassName + "'; the base locale is '" +
187                    resourceList.locale + "'.");
188            pw.println("# It was generated by " + ResourceGen.class);
189    
190            pw.println("# from " + getFileForComments());
191            if (include.root.commentStyle != 
192                    ResourceGenTask.COMMENT_STYLE_SCM_SAFE) {
193                pw.println("# on " + new Date().toString() + ".");
194            }
195            pw.println();
196            for (int i = 0; i < resourceList.resources.length; i++) {
197                ResourceDef.Resource resource = resourceList.resources[i];
198                final String name = resource.name;
199                if (resource.text == null) {
200                    throw new BuildException(
201                            "Resource '" + name + "' has no message");
202                }
203                final String message = resource.text.cdata;
204                if (message == null) {
205                    continue;
206                }
207                pw.println(name + "=" + Util.quoteForProperties(message));
208            }
209            pw.println("# End " + fullClassName + ".properties");
210        }

213        /**
214         * Generates a properties file for a given locale. If there is a source
215         * file for the locale, it is copied. Otherwise generates a file with
216         * headers but no resources.
217         *
218         * @param pw Output file writer
219         * @param targetFile the locale-specific output file
220         * @param srcFile The locale-specific properties file, e.g.
221         *   "source/happy/BirthdayResource_fr-FR.properties". It may not exist,
222         *   but if it does, we copy it.
223         * @param locale Locale, never null
224         * @pre locale != null
225         */
226        private void generateProperties(
227            PrintWriter pw,
228            File targetFile,
229            File srcFile,
230            Locale locale)
231        {
232            if (srcFile.exists() && srcFile.canRead() && !targetFile.equals(srcFile)) {
233                try {
234                    final FileReader reader = new FileReader(srcFile);
235    
236                    final char[] cbuf = new char[1000];
237                    int charsRead;
238                    while ((charsRead = reader.read(cbuf)) > 0) {
239                        pw.write(cbuf, 0, charsRead);
240                    }
241                    return;
242                } catch (IOException e) {
243                    throw new BuildException("Error while copying from '" +
244                        srcFile + "'");
245                }
246            }
247    
248            // Generate an empty file.
249            String fullClassName = getClassName(locale);
250            pw.println("# This file contains the resources for");
251            pw.println("# class '" + fullClassName + "' and locale '" + locale + "'.");
252            pw.println("# It was generated by " + ResourceGen.class);
253            pw.println("# from " + getFileForComments());
254            if (include.root.commentStyle != 
255                    ResourceGenTask.COMMENT_STYLE_SCM_SAFE) {
256                pw.println("# on " + new Date().toString() + ".");
257            }
258            pw.println();
259            pw.println("# This file is intentionally blank. Add property values");
260            pw.println("# to this file to override the translations in the base");
261            String basePropertiesFileName = Util.getClassNameSansPackage(className, locale) + ".properties";
262            pw.println("# properties file, " + basePropertiesFileName);
263            pw.println();
264            pw.println("# End " + fullClassName + ".properties");
265        }
266    
267        private String getClassName(Locale locale) {
268            String s = className;
269            if (locale != null) {
270                s += '_' + locale.toString();
271            }
272            return s;
273        }
274    
275        protected void generateCpp(
276            ResourceGen generator,
277            ResourceDef.ResourceBundle resourceList)
278        {
279            String defaultExceptionClass = resourceList.cppExceptionClassName;
280            String defaultExceptionLocation = resourceList.cppExceptionClassLocation;
281            if (defaultExceptionClass != null &&
282                defaultExceptionLocation == null) {
283                throw new BuildException(
284                    "C++ exception class is defined without a header file location in "
285                    + getFile());
286            }
287    
288            for (int i = 0; i < resourceList.resources.length; i++) {
289                ResourceDef.Resource resource = resourceList.resources[i];
290    
291                if (resource.text == null) {
292                    throw new BuildException(
293                        "Resource '" + resource.name + "' has no message");
294                }
295    
296                if (resource instanceof ResourceDef.Exception) {
297                    ResourceDef.Exception exception =
298                        (ResourceDef.Exception)resource;
299    
300                    if (exception.cppClassName != null &&
301                        (exception.cppClassLocation == null &&
302                         defaultExceptionLocation == null)) {
303                        throw new BuildException(
304                            "C++ exception class specified for "
305                            + exception.name
306                            + " without specifiying a header location in "
307                            + getFile());
308                    }
309    
310                    if (defaultExceptionClass == null &&
311                        exception.cppClassName == null) {
312                        throw new BuildException(
313                            "No exception class specified for "
314                            + exception.name
315                            + " in "
316                            + getFile());
317                    }
318                }
319            }
320    
321    
322            String hFilename = cppClassName + ".h";
323            String cppFileName = cppClassName + ".cpp";
324    
325            File hFile = new File(include.root.dest, hFilename);
326            File cppFile = new File(include.root.dest, cppFileName);
327    
328            boolean allUpToDate = true;
329    
330            if (!checkUpToDate(generator, hFile)) {
331                allUpToDate = false;
332            }
333    
334            if (!checkUpToDate(generator, cppFile)) {
335                allUpToDate = false;
336            }
337    
338            if (allUpToDate && !include.root.force) {
339                return;
340            }
341    
342            generator.comment("Generating " + hFile);
343    
344            final FileOutputStream hOut;
345            try {
346                makeParentDirs(hFile);
347    
348                hOut = new FileOutputStream(hFile);
349            } catch(FileNotFoundException e) {
350                throw new BuildException("Error while writing " + hFile, e);
351            }
352    
353            String className = Util.removePackage(this.className);
354            String baseClassName = Util.removePackage(this.cppBaseClassName);
355    
356            PrintWriter pw = new PrintWriter(hOut);
357            try {
358                final CppHeaderGenerator gen =
359                    new CppHeaderGenerator(getFile(), hFile,
360                    className, baseClassName, defaultExceptionClass);
361                configureCommentStyle(gen);
362                gen.generateModule(generator, resourceList, pw);
363            } finally {
364                pw.close();
365            }
366    
367            generator.comment("Generating " + cppFile);
368    
369            final FileOutputStream cppOut;
370            try {
371                makeParentDirs(cppFile);
372    
373                cppOut = new FileOutputStream(cppFile);
374            } catch(FileNotFoundException e) {
375                throw new BuildException("Error while writing " + cppFile, e);
376            }
377    
378            pw = new PrintWriter(cppOut);
379            try {
380                final CppGenerator gen =
381                    new CppGenerator(getFile(), cppFile, className, baseClassName,
382                        defaultExceptionClass, hFilename);
383                configureCommentStyle(gen);
384                gen.generateModule(generator, resourceList, pw);
385            } finally {
386                pw.close();
387            }
388        }
389    }
